  <div class="section-label">◈ 03 — Mathematical Model</div>
  <div class="formula-grid">
    <div class="formula-block" data-tag="ENCODING">
      <div class="formula-name">Signal synthesis</div>
      <div class="formula">
        S(<em>x</em>,<em>t</em>) = Σ<sub>k</sub> <strong>v</strong>[k] · ψ<sub>k</sub>(t) · Φ<sub>k</sub>(x)
      </div>
      <div class="formula-note">
        Φ<sub>k</sub> = 2D DCT spatial modes (lowest freq first)<br/>
        ψ<sub>k</sub> = cosine/sine temporal envelopes ≤ 4 Hz<br/>
        <strong>v</strong> = orthogonal codebook vector ∈ ℝ²⁶
      </div>
    </div>
    <div class="formula-block" data-tag="DECODING">
      <div class="formula-name">Feature extraction (matched filter)</div>
      <div class="formula">
        v̂[k] = (1/T) Σ<sub>t</sub> ψ<sub>k</sub>(t) · Φ<sub>k</sub><sup>T</sup> · <em>g</em><sub>t</sub>
      </div>
      <div class="formula-note">
        Optimal linear estimator under AWGN.<br/>
        Projects observed frames back into coefficient space.<br/>
        Then: ĉ = argmin<sub>i</sub> ‖v̂ − v<sub>i</sub>‖₂
      </div>
    </div>
    <div class="formula-block" data-tag="CODEBOOK">
      <div class="formula-name">Orthogonal codebook construction</div>
      <div class="formula">
        Q, _ = QR(<em>R</em>)  where <em>R</em> ~ 𝒩(0,1)<br/>
        v<sub>i</sub> = <strong>3</strong> · Q<sub>i</sub><br/>
        d(v<sub>i</sub>, v<sub>j</sub>) = √2 · 3 = <strong>4.24</strong>
      </div>
      <div class="formula-note">
        All 325 pairwise distances equal exactly √2 · scale.<br/>
        Maximum possible d_min for N ≤ K on unit hypersphere.
      </div>
    </div>
    <div class="formula-block" data-tag="ROBUSTNESS">
      <div class="formula-name">Noise margin analysis</div>
      <div class="formula">
        σ<sub>noise</sub> = 0.05 → ‖noise‖ = 0.255<br/>
        boundary = d_min/2 = <strong>2.121</strong><br/>
        margin = 2.121 / 0.255 = <strong>8.32×</strong>
      </div>
      <div class="formula-note">
        At σ=0.05 per-cell noise, the system has 8.32× headroom.<br/>
        Accuracy degrades only beyond σ=0.08 (strong blur/noise).
      </div>
    </div>
  </div>

const CELL = 16;
const COLORS = {
  on:  { H:'#00e5ff', E:'#39ff6e', L:'#ffb700', O:'#ff3b5c' },
  off: '#0c1117',
  border: '#1a2530'
};

function drawGrid(canvas, grid, onColor) {
  canvas.width = 10 * CELL; canvas.height = 10 * CELL;
  const ctx = canvas.getContext('2d');
  ctx.fillStyle = '#06080a';
  ctx.fillRect(0, 0, canvas.width, canvas.height);
  for (let r = 0; r < 10; r++) {
    for (let c = 0; c < 10; c++) {
      const on = grid[r][c] === 1;
      ctx.fillStyle = on ? onColor : COLORS.off;
      ctx.fillRect(c*CELL+1, r*CELL+1, CELL-2, CELL-2);
      if (on) {
        ctx.shadowColor = onColor;
        ctx.shadowBlur = 4;
        ctx.fillStyle = onColor;
        ctx.fillRect(c*CELL+1, r*CELL+1, CELL-2, CELL-2);
        ctx.shadowBlur = 0;
      }
    }
  }
}

// Draw the special control frames

drawGrid(document.getElementById('syncOff'),  GRIDS.SYNC_OFF, '#334455');
drawGrid(document.getElementById('delimCanvas'), GRIDS.DELIM, '#ffb700');

// Build character trajectory rows
const vis = document.getElementById('gridVis');
const timeLabels = ['t=0.5s', 't=2.0s', 't=3.5s'];
const chars = ['H','E','L','L','O'];

const row = document.createElement('div');
row.style.cssText = 'display:flex;gap:20px;align-items:flex-start;flex-wrap:wrap';

chars.forEach((ch, ci) => {
  const block = document.createElement('div');
  block.style.cssText = 'display:flex;flex-direction:column;align-items:center;gap:6px';

  const label = document.createElement('div');
  label.textContent = ch;
  label.style.cssText = `font-family:'Bebas Neue',sans-serif;font-size:28px;letter-spacing:3px;color:${COLORS.on[ch] || '#fff'}`;
  block.appendChild(label);

  const framesDiv = document.createElement('div');
  framesDiv.style.cssText = 'display:flex;gap:4px;align-items:center';

  const srcGrids = ch === 'L' && ci === 3 ? GRIDS['L'] : GRIDS[ch];

  srcGrids.forEach((grid, fi) => {
    const wrap = document.createElement('div');
    const c = document.createElement('canvas');
    drawGrid(c, grid, COLORS.on[ch] || '#00e5ff');
    c.style.cssText = 'display:block;image-rendering:pixelated;border:1px solid #1a2530';
    const tl = document.createElement('div');
    tl.textContent = timeLabels[fi];
    tl.style.cssText = `font-family:'Space Mono',monospace;font-size:8px;color:#4a5568;text-align:center;margin-top:3px`;
    wrap.appendChild(c); wrap.appendChild(tl);
    if (fi < 2) {
      const arr = document.createElement('span');
      arr.textContent = '→';
      arr.style.cssText = 'font-family:monospace;font-size:12px;color:#1a2530;align-self:flex-start;margin-top:5px';
      framesDiv.appendChild(wrap);
      framesDiv.appendChild(arr);
    } else {
      framesDiv.appendChild(wrap);
    }
  });

  block.appendChild(framesDiv);
  row.appendChild(block);
});
vis.appendChild(row);
